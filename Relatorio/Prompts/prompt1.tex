\section{Definição do Domínio do Sistema}

\textbf{Domínio:} Gestão Integrada de Clube Desportivo [cite: 38]

O sistema é destinado à gestão completa de um clube desportivo, academia ou associação que necessite de: [cite: 38]
\begin{itemize}
    \item Registo e gestão de atletas [cite: 38]
    \item Organização de equipas e atribuição de modalidades [cite: 38]
    \item Gestão de treinadores e seus horários [cite: 38]
    \item Controlo de inscrições e presenças em treinos [cite: 41]
    \item Participação em competições e registo de resultados [cite: 41]
    \item Análise de desempenho e estatísticas [cite: 41]
\end{itemize}

\section{User Stories}

\textbf{US-1: Registar Novo Atleta}
\begin{verbatim}
Como secretária, pretendo registar um novo atleta no sistema
para manter a base de dados do clube atualizada. [cite: 38, 61]
Critérios de Aceitação:
- Inserir dados pessoais (nome, data de nascimento, etc.)
- Validar email e telefone
- Atribuir modalidade
\end{verbatim}

\textbf{US-2: Consultar Desempenho de Atleta}
\begin{verbatim}
Como treinador, pretendo consultar o desempenho de um atleta
para avaliar sua progressão e necessidade de ajustes. [cite: 41, 61]
Critérios de Aceitação:
- Ver estatísticas de competições
- Visualizar frequência de treinos
- Gráficos de evolução temporal
\end{verbatim}

\begin{equation*}

\{id, nome, dataInicio, dataFim, tipo, modalidade\}

\textbf{Resultado}
\begin{equation*}
\{id, \text{competição}\_id, equipa\_id, \text{posição}, pontos, data\}
\end{equation*}

\textbf{Quota}
\begin{equation*}
\{id, atleta\_id, mesReferente, valor, dataPagamento, status\}
\end{equation*}
